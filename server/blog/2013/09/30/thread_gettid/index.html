#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>
#include <unistd.h>

#include <sys/syscall.h>  
#define gettid() syscall(__NR_gettid)

void *hello(void *ptr)
{
    while(1)
    {   
        printf("%lx %ld\n",pthread_self(),gettid());
        sleep(1);   
    }   
}

int main()
{
    pthread_t thread_;
    pthread_create(&thread_, NULL, hello, NULL);  
    while(1)
    {   
        printf("%lx %ld\n",pthread_self(),gettid());
        sleep(1);                                                                                                                          
    }   
    return 0;
}
