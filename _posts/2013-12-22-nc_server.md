在老版本中，有-c命令直接执行命令，新版本仍然可以通过其他方式。

##Server

	#include <stdio.h>
	int main()
	{
		char buf_[1024];
		do{
			if(gets(buf_))
			{
				printf("recv : %s\n",buf_);
				fflush(stdout);
			}
		}while(1);
		return 0;
	}
